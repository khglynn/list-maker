Soundiiz accepts a CSV with headers that look like this (order can vary but the key columns are Title, Artist, and Album):

```
Title,Artist,Album,Disc Number,Track Number,Year
```

## 4) Example Python Code

Here’s a rough skeleton you could adapt. It won’t work exactly as-is until you adjust the selectors/logic for the actual site layout, but it should give you a starting point:

```python
import requests
from bs4 import BeautifulSoup
import csv
import time

BASE_URL = "https://switchedonpop.com"
EPISODES_URL = "https://switchedonpop.com/episodes"

def get_episode_links(page=1):
    """
    Fetch the episodes listing page and return a list of episode detail URLs
    for that page. Adjust logic for however the site lists episodes.
    """
    # Potentially there's a query param like ?page={page}
    url = f"{EPISODES_URL}?page={page}"
    r = requests.get(url)
    soup = BeautifulSoup(r.text, "html.parser")
    
    # Adjust this based on actual HTML structure of episodes listing
    episode_cards = soup.find_all("div", class_="episode-card") 
    links = []
    for card in episode_cards:
        link_tag = card.find("a")
        if link_tag and "href" in link_tag.attrs:
            links.append(BASE_URL + link_tag["href"])
    return links

def get_songs_discussed(episode_url):
    """
    Given a single episode's URL, fetch the page and parse out the 'Songs Discussed.'
    Returns a list of (Title, Artist) tuples (or dicts).
    """
    r = requests.get(episode_url)
    soup = BeautifulSoup(r.text, "html.parser")
    
    # This is TOTALLY dependent on how the site is structured.
    # For example, it might be a <h2>Songs Discussed</h2> 
    # followed by a <ul> or <div>.
    
    # Let's pretend there's a <h2>Songs Discussed</h2> 
    # and the next sibling is a <ul> with <li>Song Title – Artist</li>.
    
    songs_list = []
    
    # Find the header
    songs_header = soup.find("h2", string=lambda t: t and "Songs Discussed" in t)
    if songs_header:
        # Maybe the next element is a UL or DIV
        potential_list = songs_header.find_next_sibling(["ul","div"])
        if potential_list:
            list_items = potential_list.find_all("li")
            for li in list_items:
                # The site often uses "Title – Artist" or "Title, Artist" style
                # You might have to do custom parsing.
                
                text = li.get_text(strip=True)
                # Basic approach: split on " – " (space-dash-space)
                if "–" in text:
                    title_part, artist_part = text.split("–", 1)
                    title = title_part.strip('“”"\' ')
                    artist = artist_part.strip()
                    songs_list.append((title, artist))
                else:
                    # handle any lines that don't follow the pattern
                    songs_list.append((text, ""))
    
    return songs_list

def scrape_all_songs(pages_to_scrape=20):
    """
    Scrape all episodes from page 1 to `pages_to_scrape`,
    then gather the 'Songs Discussed' for each episode.
    """
    all_song_rows = []
    for p in range(1, pages_to_scrape + 1):
        print(f"Scraping page {p}...")
        episode_links = get_episode_links(page=p)
        if not episode_links:
            print("No more episodes found.")
            break
        
        for link in episode_links:
            print(f"  Scraping episode: {link}")
            songs = get_songs_discussed(link)
            # We can store the episode link or parse the page's title for reference
            # to keep track of which episode they belong to.
            
            # Attempt to find episode title from <title> or a heading
            # This depends on the site structure.
            ep_title_tag = None
            ep_title = ""
            
            try:
                # Quick fetch just for the title (or reuse the soup from get_songs_discussed)
                # If we want to reuse soup, we can restructure the code.
                r = requests.get(link)
                soup = BeautifulSoup(r.text, "html.parser")
                ep_title_tag = soup.find("h1", class_="episode-title")  # Example
                if ep_title_tag:
                    ep_title = ep_title_tag.get_text(strip=True)
            except:
                pass
            
            # Now build the rows for the CSV
            for title, artist in songs:
                # For simplicity: Title,Artist,Album,Disc Number,Track Number,Year,Episode
                all_song_rows.append({
                    "Title": title,
                    "Artist": artist,
                    "Album": "",
                    "Disc Number": "",
                    "Track Number": "",
                    "Year": "",
                    "Episode": ep_title if ep_title else link
                })
            
            # throttle requests
            time.sleep(1)
    
    return all_song_rows

def main():
    # Adjust pages_to_scrape to cover how many pages you think exist
    # If each page is 20 episodes and you need ~390 episodes total,
    # you might set this to 20 or so (depending on how many total pages the site has).
    all_songs = scrape_all_songs(pages_to_scrape=20)
    
    # Write to CSV
    with open("SwitchedOnPop.csv", mode="w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(
            f,
            fieldnames=["Title", "Artist", "Album", "Disc Number", "Track Number", "Year", "Episode"]
        )
        writer.writeheader()
        for row in all_songs:
            writer.writerow(row)

if __name__ == "__main__":
    main()
```
